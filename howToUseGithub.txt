
# Clone the repository to your computer

	git clone [INSERT REPOSITORY ADDRESS]

# View the status of all files in your repository (see files that still need to be committed).
	
	git status

# Add file/directory contents to the index so they can be committed.

	git add [INSERT FILES/DIRECTORIES HERE]

# Used to add a message about what files you are committing to the repository.

	git commit -m "[ADD MESSAGE ABOUT THE FILE(S) BEING COMMITTED/UPDATED HERE]"

# Used to actually push (publish) the changes to the repository

	git push

# ======================================================================
# Pull updates others may have made to the repository

	git pull


# ======================================================================
# Remove deleted files from tracking

	git add -u .
	git commit -m "removing deleted files from tracking"
	git push
